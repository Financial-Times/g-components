# Election utilitity functions

`getPartyInfo(string)` returns an object with party `color`, `shortName` and
`formattedName` for a given UK political party.

<Preview>
  <Story name="getPartyInfo(party: string)">
    <div>
      {parties.map(p => {
        const { color, shortName, formattedName, adjective } = getPartyInfo(p);
        return (
          <div>
            <div
              style={{ height: 20, width: 20, backgroundColor: color, border: '1px solid black' }}
            />
            <div>
              <strong>shortName: </strong>
              {`${shortName}`}
            </div>
            <div>
              <strong>formattedName: </strong>
              {`${formattedName}`}
            </div>
            <div>
              <strong>adjective: </strong>
              {`e.g. the ${adjective || formattedName} candidate`}
            </div>
            <br />
          </div>
        );
      })}
    </div>
